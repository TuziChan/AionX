# 04 - 前端重构设计

## 1. 重构目标

- 完全复刻原有界面和交互体验
- 消除重复的 Agent 聊天组件，提取统一抽象
- 消除重复的设置页面模式，建立通用框架
- 用 Zustand 替代 Context 三件套，减少样板代码
- 将 Electron IPC 调用替换为 Tauri invoke
- 保持 Arco Design + UnoCSS 样式方案不变

## 2. 核心重构：统一 Agent 聊天架构

### 2.1 现有问题

原项目中每个 Agent 都有独立的聊天组件：

```
pages/conversation/
├── acp/AcpChat.tsx          # ACP 聊天（独立实现）
├── acp/AcpSendBox.tsx       # ACP 发送框（独立实现）
├── codex/CodexChat.tsx      # Codex 聊天（独立实现）
├── gemini/GeminiChat.tsx    # Gemini 聊天（独立实现）

├── openclaw/OpenClawChat.tsx # OpenClaw 聊天（独立实现）
```

这些组件 80%+ 的逻辑是相同的（消息列表、输入框、流式显示、工具调用等），只有少量 Agent 特有的差异。

interface SettingsPageProps {
  category: string;
  sections: SettingsSection[];

export function SettingsPage({ category, sections }: SettingsPageProps) {
  const { settings, updateSettings, isLoading } = useSettings(category);
  const { t } = useTranslation();

  return (
    <div className="p-6 max-w-3xl mx-auto">
      {sections.map((section) => (
        <div key={section.key} className="mb-8">
          <h3 className="text-lg font-medium mb-2">{t(section.titleKey)}</h3>
          {section.descriptionKey && (
            <p className="text-t-secondary mb-4">{t(section.descriptionKey)}</p>
          )}
          <div className="space-y-4">
            {section.fields
              .filter((f) => !f.visible || f.visible(settings))
              .map((field) => (
                <SettingsField
                  key={field.key}
                  field={field}
                  value={settings[field.key] ?? field.defaultValue}
                  onChange={(v) => updateSettings({ [field.key]: v })}
                />
              ))}
          </div>
        </div>
      ))}
    </div>
  );
}

  const [theme, setTheme] = useState<'light' | 'dark'>('light');

  const [fontScale, setFontScale] = useState(1);

### 4.2 Store 设计

```typescript
// stores/conversation.ts
export const useConversationStore = create<ConversationStore>((set, get) => ({
  conversations: [],
  currentId: null,
  isLoading: false,

  setCurrentId: (id) => set({ currentId: id }),
  fetchConversations: async () => {
    set({ isLoading: true });
    const conversations = await invoke('list_conversations');
    set({ conversations, isLoading: false });
  },
  createConversation: async (input) => {
    const conv = await invoke('create_conversation', input);
    set((s) => ({ conversations: [conv, ...s.conversations] }));
    return conv;
  },
  deleteConversation: async (id) => {
    await invoke('delete_conversation', { id });
    set((s) => ({
      conversations: s.conversations.filter((c) => c.id !== id),
    }));
  },
}));

// stores/tabs.ts
export const useTabsStore = create<TabsStore>((set) => ({
  tabs: [],
  activeTabId: null,

  openTab: (conversation) => set((s) => {
    if (s.tabs.find((t) => t.id === conversation.id)) {
      return { activeTabId: conversation.id };
    }
    return {
      tabs: [...s.tabs, { id: conversation.id, ...conversation }],
      activeTabId: conversation.id,
    };
  }),
  closeTab: (id) => set((s) => {
    const tabs = s.tabs.filter((t) => t.id !== id);
    const activeTabId = s.activeTabId === id
      ? tabs[tabs.length - 1]?.id ?? null
      : s.activeTabId;
    return { tabs, activeTabId };
  }),
  switchTab: (id) => set({ activeTabId: id }),
}));

// stores/theme.ts
export const useThemeStore = create<ThemeStore>()(
  persist(
    (set) => ({
      theme: 'light' as 'light' | 'dark',
      colorScheme: 'default',
      fontScale: 1,
      customCss: '',

      setTheme: (theme) => set({ theme }),
      setColorScheme: (scheme) => set({ colorScheme: scheme }),
      setFontScale: (scale) => set({ fontScale: scale }),
      setCustomCss: (css) => set({ customCss: css }),
    }),
    { name: 'theme-storage' }
  )
);

// stores/auth.ts
export const useAuthStore = create<AuthStore>((set) => ({
  user: null,
  status: 'checking' as AuthStatus,

  login: async (params) => {
    const result = await invoke('login', params);
    set({ user: result.user, status: 'authenticated' });
    return result;
  },
  logout: async () => {
    await invoke('logout');
    set({ user: null, status: 'unauthenticated' });
  },
  checkAuth: async () => {
    try {
      const user = await invoke('get_current_user');
      set({ user, status: 'authenticated' });
    } catch {
      set({ user: null, status: 'unauthenticated' });
    }
  },
}));

// stores/settings.ts
export const useSettingsStore = create<SettingsStore>((set, get) => ({
  cache: {} as Record<string, any>,

  getSettings: async (category) => {
    if (get().cache[category]) return get().cache[category];
    const settings = await invoke('get_settings', { category });
    set((s) => ({ cache: { ...s.cache, [category]: settings } }));
    return settings;
  },
  updateSettings: async (category, updates) => {
    await invoke('update_settings', { category, settings: updates });
    set((s) => ({
      cache: {
        ...s.cache,
        [category]: { ...s.cache[category], ...updates },
      },
    }));
  },
}));
```

## 5. 服务层重构：IPC Bridge → Tauri Invoke

### 5.1 统一 API 封装

```typescript
// services/api.ts
import { invoke } from '@tauri-apps/api/core';
import { listen, type UnlistenFn } from '@tauri-apps/api/event';

/** 类型安全的 invoke 封装 */
export async function api<T>(command: string, args?: Record<string, unknown>): Promise<T> {
  try {
    return await invoke<T>(command, args);
  } catch (error) {
    // 统一错误处理
    throw new ApiError(command, error);
  }
}

/** 事件监听封装 */
export function onEvent<T>(event: string, handler: (payload: T) => void): Promise<UnlistenFn> {
  return listen<T>(event, (e) => handler(e.payload));
}
```

### 6.2 重构方案：消息渲染注册表

```typescript
// components/message/registry.ts

type MessageRenderer = React.ComponentType<{ message: Message }>;

const renderers = new Map<string, MessageRenderer>();

export function registerRenderer(type: string, renderer: MessageRenderer) {
  renderers.set(type, renderer);
}

export function getRenderer(message: Message): MessageRenderer {
  // 优先匹配 agent_type + message_type
  const specific = renderers.get(`${message.agentType}:${message.type}`);
  if (specific) return specific;

  // 回退到通用 message_type
  const generic = renderers.get(message.type);
  if (generic) return generic;

  // 默认文本渲染
  return DefaultMessageRenderer;
}

// 注册通用渲染器
registerRenderer('text', TextMessage);
registerRenderer('tool_call', ToolCallMessage);
registerRenderer('plan', PlanMessage);
registerRenderer('permission', PermissionMessage);
registerRenderer('file_changes', FileChangesMessage);
registerRenderer('status', StatusMessage);
registerRenderer('tips', TipsMessage);

// 注册 Agent 特有渲染器（仅在确实有差异时）
registerRenderer('acp:permission', AcpPermissionMessage);
```
