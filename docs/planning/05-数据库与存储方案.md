# 05 - 数据库与存储方案

## 1. 数据库技术选型

| 对比项 | 原方案 (better-sqlite3) | 新方案 (rusqlite) |
|--------|------------------------|-------------------|
| 语言 | Node.js 原生绑定 | Rust 原生 |
| 模式 | 同步 API | 同步 + 异步（tokio 包装） |
| 连接池 | 无（单连接） | r2d2 连接池 |
| 性能 | 优秀 | 更优（无 FFI 开销） |
| 迁移 | 手动版本管理 | refinery 或手动 |
| 打包 | 需要 prebuild | bundled 特性，零依赖 |

```rust
// src-tauri/src/db/repositories/conversation.rs

pub struct ConversationRepository {
    pool: Arc<Pool<SqliteConnectionManager>>,
}

impl ConversationRepository {
    pub fn create(&self, input: &CreateConversation, user_id: Option<&str>) -> Result<Conversation, AppError> {
        let conn = self.pool.get()?;
        let id = uuid::Uuid::new_v4().to_string();
        conn.execute(
            "INSERT INTO conversations (id, user_id, name, agent_type, model, workspace_path, extra)
             VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7)",
            params![
                id,
                user_id,
                input.name,
                input.agent_type,
                input.model.as_deref().unwrap_or(""),
                input.workspace_path.as_deref().unwrap_or(""),
                serde_json::to_string(&input.extra.clone().unwrap_or_default())?,
            ],
        )?;
        self.get(&id)
    }

    pub fn get(&self, id: &str) -> Result<Conversation, AppError> {
        let conn = self.pool.get()?;
        conn.query_row(
            "SELECT * FROM conversations WHERE id = ?1",
            params![id],
            |row| Ok(Conversation::from_row(row)),
        ).map_err(|_| AppError::NotFound(format!("Conversation {id}")))
    }

    pub fn list(&self, params: &ListParams) -> Result<PaginatedResult<Conversation>, AppError> {
        let conn = self.pool.get()?;
        let offset = (params.page.unwrap_or(1) - 1) * params.page_size.unwrap_or(50);
        let limit = params.page_size.unwrap_or(50);

        let total: i64 = conn.query_row(
            "SELECT COUNT(*) FROM conversations WHERE status != 'deleted'",
            [],
            |row| row.get(0),
        )?;

        let mut stmt = conn.prepare(
            "SELECT * FROM conversations WHERE status != 'deleted'
             ORDER BY updated_at DESC LIMIT ?1 OFFSET ?2"
        )?;
        let items = stmt.query_map(params![limit, offset], |row| {
            Ok(Conversation::from_row(row))
        })?.collect::<Result<Vec<_>, _>>()?;

        Ok(PaginatedResult { items, total, page: params.page.unwrap_or(1), page_size: limit })
    }

    pub fn update(&self, id: &str, updates: &ConversationUpdate) -> Result<Conversation, AppError> {
        let conn = self.pool.get()?;
        // 动态构建 UPDATE 语句
        let mut sets = vec!["updated_at = unixepoch()".to_string()];
        let mut values: Vec<Box<dyn rusqlite::ToSql>> = vec![];

        if let Some(ref name) = updates.name {
            sets.push(format!("name = ?{}", values.len() + 1));
            values.push(Box::new(name.clone()));
        }
        if let Some(ref model) = updates.model {
            sets.push(format!("model = ?{}", values.len() + 1));
            values.push(Box::new(model.clone()));
        }
        // ... 其他字段

        let sql = format!("UPDATE conversations SET {} WHERE id = ?{}", sets.join(", "), values.len() + 1);
        values.push(Box::new(id.to_string()));
        conn.execute(&sql, rusqlite::params_from_iter(values.iter().map(|v| v.as_ref())))?;

        self.get(id)
    }

    pub fn delete(&self, id: &str) -> Result<(), AppError> {
        let conn = self.pool.get()?;
        conn.execute("DELETE FROM conversations WHERE id = ?1", params![id])?;
        Ok(())
    }

    pub fn get_by_workspace(&self, path: &str) -> Result<Vec<Conversation>, AppError> {
        let conn = self.pool.get()?;
        let mut stmt = conn.prepare(
            "SELECT * FROM conversations WHERE workspace_path = ?1 ORDER BY updated_at DESC"
        )?;
        let items = stmt.query_map(params![path], |row| {
            Ok(Conversation::from_row(row))
        })?.collect::<Result<Vec<_>, _>>()?;
        Ok(items)
    }
```

## 5. 数据库连接池

```rust
// src-tauri/src/db/mod.rs
use r2d2::Pool;
use r2d2_sqlite::SqliteConnectionManager;

pub type DatabasePool = Pool<SqliteConnectionManager>;

pub fn init_database(app_data_dir: &std::path::Path) -> Result<DatabasePool, AppError> {
    let db_path = app_data_dir.join("aionui.db");

    let manager = SqliteConnectionManager::file(&db_path)
        .with_init(|conn| {
            conn.execute_batch(
                "PRAGMA journal_mode = WAL;
                 PRAGMA foreign_keys = ON;
                 PRAGMA busy_timeout = 5000;
                 PRAGMA synchronous = NORMAL;
                 PRAGMA cache_size = -64000;  -- 64MB
                 PRAGMA temp_store = MEMORY;"
            )?;
            Ok(())
        });

    let pool = Pool::builder()
        .max_size(10)
        .build(manager)?;

    // 运行迁移
    run_migrations(&pool)?;

    Ok(pool)
}
```

## 6. 迁移系统

```rust
// src-tauri/src/db/migration.rs

struct Migration {
    version: u32,
    description: &'static str,
    sql: &'static str,
}

const MIGRATIONS: &[Migration] = &[
    Migration {
        version: 1,
        description: "Initial schema",
        sql: include_str!("../../migrations/001_initial.sql"),
    },
    Migration {
        version: 2,
        description: "Add cron jobs",
        sql: include_str!("../../migrations/002_cron_jobs.sql"),
    },
    Migration {
        version: 3,
        description: "Add MCP servers",
        sql: include_str!("../../migrations/003_mcp_servers.sql"),
    },
    Migration {
        version: 4,
        description: "Add extensions",
        sql: include_str!("../../migrations/004_extensions.sql"),
    },
];

pub fn run_migrations(pool: &DatabasePool) -> Result<(), AppError> {
    let conn = pool.get()?;

    // 创建迁移跟踪表
    conn.execute_batch(
        "CREATE TABLE IF NOT EXISTS _migrations (
            version INTEGER PRIMARY KEY,
            description TEXT NOT NULL,
            applied_at INTEGER NOT NULL DEFAULT (unixepoch())
        );"
    )?;

    let current_version: u32 = conn
        .query_row("SELECT COALESCE(MAX(version), 0) FROM _migrations", [], |row| row.get(0))
        .unwrap_or(0);

    for migration in MIGRATIONS {
        if migration.version > current_version {
            tracing::info!("Running migration v{}: {}", migration.version, migration.description);
            conn.execute_batch(migration.sql)?;
            conn.execute(
                "INSERT INTO _migrations (version, description) VALUES (?1, ?2)",
                params![migration.version, migration.description],
            )?;
        }
    }

    Ok(())
}
```
